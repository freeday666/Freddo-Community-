// ================= DISCORD PANEL V2 ================= // Stack: Express + SQLite + JWT + discord.js // Features: // - Auth (JWT) // - Database (SQLite) // - Encrypted tokens // - Dashboard UI // - Hosting ready // ================= INSTALL ================= // npm init -y // npm install express discord.js cors sqlite3 jsonwebtoken bcrypt // ================= SERVER ================= const express = require('express'); const { Client, GatewayIntentBits } = require('discord.js'); const cors = require('cors'); const sqlite3 = require('sqlite3').verbose(); const jwt = require('jsonwebtoken'); const bcrypt = require('bcrypt'); const app = express(); app.use(cors()); app.use(express.json()); const SECRET = "supersecret"; const db = new sqlite3.Database('./database.db'); // Create tables db.run(`CREATE TABLE IF NOT EXISTS users ( id INTEGER PRIMARY KEY, username TEXT, password TEXT )`); db.run(`CREATE TABLE IF NOT EXISTS bots ( id INTEGER PRIMARY KEY, token TEXT )`); // Register app.post('/register', async (req, res) => { const { username, password } = req.body; const hash = await bcrypt.hash(password, 10); db.run("INSERT INTO users (username, password) VALUES (?, ?)", [username, hash]); res.json({ success: true }); }); // Login app.post('/login', (req, res) => { const { username, password } = req.body; db.get("SELECT * FROM users WHERE username = ?", [username], async (err, user) => { if (!user) return res.json({ error: 'User niet gevonden' }); const valid = await bcrypt.compare(password, user.password); if (!valid) return res.json({ error: 'Fout wachtwoord' }); const token = jwt.sign({ id: user.id }, SECRET); res.json({ token }); }); }); // Middleware function auth(req, res, next) { const token = req.headers.authorization; if (!token) return res.sendStatus(403); try { req.user = jwt.verify(token, SECRET); next(); } catch { res.sendStatus(403); } } let runningBots = []; // Start bot app.post('/start', auth, async (req, res) => { const { token } = req.body; try { const client = new Client({ intents: [GatewayIntentBits.Guilds] }); await client.login(token); runningBots.push(client); db.run("INSERT INTO bots (token) VALUES (?)", [token]); res.json({ success: true }); } catch { res.json({ error: 'Token fout' }); } }); // Stop bot app.post('/stop', auth, (req, res) => { const bot = runningBots.pop(); if (bot) bot.destroy(); res.json({ success: true }); }); // Status app.get('/status', auth, (req, res) => { res.json({ online: runningBots.length }); }); app.listen(3000, () => console.log('V2 server running')); // ================= FRONTEND ================= /* Panel V2
